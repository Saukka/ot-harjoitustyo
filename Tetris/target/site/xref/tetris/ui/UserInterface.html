2   package tetris.ui;
3   
4   import javafx.scene.layout.Pane;
5   import javafx.scene.control.Label;
6   import javafx.application.Application;
7   import javafx.stage.Stage;
8   import javafx.scene.control.Button;
9   import javafx.scene.Scene;
10  import tetris.domain.Tetris;
11   
12  public class UserInterface extends Application {
13      
14      @Override
15      public void start(Stage window) {
16          
17          window.setTitle("Tetris");
18          
19          Button startButton = new Button("Start");
20          startButton.setLayoutX(180);
21          startButton.setLayoutY(140);
22          
23          Label controls = new Label("Controls: \n Rotate left Z \n Hard drop SPACE \n Move piece LEFT, RIGHT");
24          controls.setLayoutX(30);
25          controls.setLayoutY(50);
26                  
27          Pane setUp = new Pane();
28          setUp.getChildren().addAll(startButton,controls);
29          
30          Scene scene = new Scene(setUp,400,300);
31          
32          window.setScene(scene);
33          
34          Tetris tetris = new Tetris();
35          
36          
37          startButton.setOnAction(e-> {
38              tetris.start(window);
39          });
40          
41          window.show();   
42      }
43      
44      public static void main(String[] args) {
45          launch(args);
46      }
47      
48  }
